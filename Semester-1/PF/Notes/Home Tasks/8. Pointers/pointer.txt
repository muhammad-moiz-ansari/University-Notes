char* FUN(char* X, char* Y, int m, int n)
{
    char* L = new char[m*(n+1)];
    int i, j;
    for (i = 0; i < m-1; i++) {
        for (j = 0; j <= n; j++) {
          if (i == 0 && j == 0)
                *(L + (i*m)+j) = Y[i] ;
          else if (i == 0 || j == 0)
                *(L + (i*m)+j) = X[j] > Y[i] ? X[j] : Y[i];
          else if (X[i - 1] == Y[j - 1])
                *(L + (i*m)+j) = X[i - 1] + 1;
          else
                *(L + i*m+j) = X[i-1] < Y[i-1] ? X[i-1] : Y[i-1];
        }
        *(L + (i*m)+j) = '\0';
    }
   
  return L;
}
void display(char *ptr,int m,int n){
  for(int i=0;i<m+1;i++)
  {
    cout<<ptr<<endl;
    ptr+=n;
  }

}
int main()
{
