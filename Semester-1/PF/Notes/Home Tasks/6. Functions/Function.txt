int power(int b,int p)
{
    int result=1,count=1;
    while(count++<=p)
        result*=b;
    
    
    return result;
}
int length(int N)
{
    int l=0;
    if(N>0 && N<9)
        return 1;
    
    while(N!=0)
    {
        l+=1;
        N/=10;
        
    }
    return l;
}
int sum(int firstOperand , int secondOperand)
{
     return firstOperand + secondOperand;
}
bool isArmstrongNumber(int N)
{
    
    int l=length(N),s=0,tempN=N;
    
    for(int i=0;i<l;i++,tempN/=10)
        {
            
            s= sum(s, power(tempN%10,l));
            
        }
   
    return  (N==s)? true : false;
}
